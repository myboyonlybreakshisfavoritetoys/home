Exercise 13: Linked-List Misarablés

Implementation of a singly-linked list with the following required operations:

  • int at(list *l, int p) - Return item at position p
  • int set(list *l, int x, int p) - Replace item at p with x
  • int size() - Return list size
  • void display() - Print list elements space-separated

Input Logic

Read t test cases. For each case, read n integers to initialize the list, then an operation code c (1-6):

  1. Append
  2. Insert Front
  3. Insert at Position
  4. Erase at Position
  5. At Position
  6. Set at Position
list.h 
typedef struct node {
    int item;
    struct node *next;
} node;

typedef struct {
    int size;
    node *head, *tail;
} list;

void initList(list*);
void push_back(list*,int);
void push_front(list*,int);
int insert(list*,int,int);
int size(list);
void display(list);
int erase(list*,int);
int at(list*,int);
int set(list*,int,int);
void freeList(list*);
list.c 

#include 
#include 
#include "list.h"

void initList(list *l){
    l->size = 0;
    l->head = l->tail = NULL;
}

void push_back(list *l, int x){
    node *n = malloc(sizeof(node));
    n->item = x;
    n->next = NULL;

    if(l->head == NULL)
        l->head = l->tail = n;

    else{
        l->tail->next = n;
        l->tail = n;
    }
    l->size++;
}

void push_front(list *l, int x){
    node *n = malloc(sizeof(node));
    n->item = x;
    n->next = NULL;

    if(l->head == NULL)
        l->head = l->tail = n;

    else{
        n->next = l->head;
        l->head = n;
    }
    l->size++;
}

int insert(list *l, int x, int pos){
    if(pos < 0 || pos > l->size)
        return 0;
    else{
        if(pos==0)
            push_front(l,x);
        else
        if(pos==l->size)
            push_back(l,x);
        else{
            node *n = malloc(sizeof(node));
            n->item = x;
            n->next = NULL;

            node *tmp = l->head;
            int p = 0;
            while(p < pos-1){
                tmp = tmp->next;
                p++;
            }
            n->next = tmp->next;
            tmp->next = n;
            l->size++;
        }
        return 1;
    }
}

int erase(list* l,int p){
   
    if (p < 0 || p >= l->size)
        return 0;

    node* toRemove;

    if (p == 0) {
        toRemove = l->head;
        l->head = toRemove->next;

       
        if (l->size == 1)
            l->tail = NULL;
    }

    else {
        node* tmp = l->head;

        for (int i = 0; i < p - 1; i++)
            tmp =  tmp->next;

        node* prev = tmp;
        toRemove = prev->next;

        prev->next = toRemove->next;

     
        if (toRemove == l->tail)
            l->tail = prev;
    }

    free(toRemove);
    l->size--;

    return 1;
}

int at(list *l, int pos) {
    if (pos < 0 || pos >= l->size) {
        return -1; 
    }

    node *tmp = l->head;
    for (int i = 0; i < pos; i++) {
        tmp = tmp->next;
    }
    
    return tmp->item;
}

int set(list *l, int x, int pos) {
    if (pos < 0 || pos >= l->size) {
        return 0; 
    }

    node *tmp = l->head;
    for (int i = 0; i < pos; i++) {
        tmp = tmp->next;
    }
    
    tmp->item = x;
    return 1; 
}

int size(list l) {
    return l.size;
}

void display(list l) {
    
    if (l.head == NULL) return;
    
    node *tmp = l.head;
    
    while (tmp != NULL) {
        printf("%d", tmp->item);
        if (tmp->next != NULL) {
            printf(" "); 
        }
        tmp = tmp->next;
    }
    printf("\n"); 
}

void freeList(list *l) {
    node *tmp = l->head;
    
    while (tmp != NULL) {
        node *next_node = tmp->next; 
        free(tmp);                   
        tmp = next_node;             
    }
    
    l->head = NULL;
    l->tail = NULL;
    l->size = 0;
}
